for k in K:
    kmeanModel = KMeans(n_clusters=k,init = 'k-means++', random_state = 42)

    distortions.append(kmeanModel.inertia_)

#Silhoutte method
ss = []
for n_cluster in range(2, 11):
    kmeans = KMeans(n_clusters=n_cluster).fit(pcX)
    label = kmeans.labels_
    sil_coeff = silhouette_score(pcX, label, metric='euclidean')
    print("For n_clusters={}, The Silhouette Coefficient is {}".format(n_cluster, sil_coeff))
    silhouette_avg = silhouette_score(x1, label)
    ss.append(silhouette_avg)

fig = plt.figure(figsize=(14,7)) 
fig.add_subplot(122)
plt.plot(K, ss,'b-',label='Silhouette Score')
plt.xlabel("Cluster Number")
plt.ylabel("Silhouette Score")
plt.legend()
plt.show()
